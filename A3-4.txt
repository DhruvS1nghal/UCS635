//Question 1
#include <iostream>
#include <cstdlib>
#include <chrono>
#include <omp.h>   
using namespace std;
#define N 1000
double get_time() {
    return chrono::duration<double>(
        chrono::high_resolution_clock::now().time_since_epoch()
    ).count();
}
int main() {
    double **A, **B, **C;
    double start, end;
    A = new double*[N];
    B = new double*[N];
    C = new double*[N];
    for (int i = 0; i < N; i++) {
        A[i] = new double[N];
        B[i] = new double[N];
        C[i] = new double[N];
    }
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++) {
            A[i][j] = 1.0;
            B[i][j] = 1.0;
            C[i][j] = 0.0;
        }
    start = get_time();
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            for (int k = 0; k < N; k++)
                C[i][j] += A[i][k] * B[k][j];
    end = get_time();
    cout << "Serial Time: " << end - start << " seconds\n";
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            C[i][j] = 0.0;
    start = get_time();
    #pragma omp parallel for schedule(static)
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            for (int k = 0; k < N; k++)
                C[i][j] += A[i][k] * B[k][j];
    end = get_time();
    cout << "OpenMP Static Time: " << end - start << " seconds\n";
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            C[i][j] = 0.0;
    start = get_time();
    #pragma omp parallel for schedule(dynamic)
    for (int i = 0; i < N; i++)
        for (int j = 0; j < N; j++)
            for (int k = 0; k < N; k++)
                C[i][j] += A[i][k] * B[k][j];
    end = get_time();
    cout << "OpenMP Dynamic Time: " << end - start << " seconds\n";
    return 0;
}

// Question 2
#include <iostream>
#include <omp.h>
using namespace std;
int main() {
    double start2, end2, start4, end4;

    #pragma omp parallel sections
    {
        #pragma omp section
        {
            start2 = omp_get_wtime();
            cout << "Series of 2 (Thread " << omp_get_thread_num() << "): ";
            for (int i = 2; i <= 20; i += 2)
                cout << i << " ";
            cout << endl;
            end2 = omp_get_wtime();
        }
        #pragma omp section
        {
            start4 = omp_get_wtime();
            cout << "Series of 4 (Thread " << omp_get_thread_num() << "): ";
            for (int i = 4; i <= 40; i += 4)
                cout << i << " ";
            cout << endl;
            end4 = omp_get_wtime();
        }
    }
    cout << "Time for series of 2: " << end2 - start2 << " seconds\n";
    cout << "Time for series of 4: " << end4 - start4 << " seconds\n";
    return 0;
}

// Question 3
#include <iostream>
#include <omp.h>
using namespace std;
#define N 10000000  
int main() {
    int *arr = new int[N];
    double start, end;
    for (int i = 0; i < N; i++)
        arr[i] = 1;
    long long total_sum;
    total_sum = 0;
    start = omp_get_wtime();
    #pragma omp parallel for
    for (int i = 0; i < N; i++) {
        total_sum += arr[i];  
    }
    end = omp_get_wtime();
    cout << "Unsynchronized Sum: " << total_sum
         << " | Time: " << end - start << " sec\n";
    total_sum = 0;
    start = omp_get_wtime();
    #pragma omp parallel for
    for (int i = 0; i < N; i++) {
        #pragma omp atomic
        total_sum += arr[i];
    }
    end = omp_get_wtime();
    cout << "Atomic Sum: " << total_sum
         << " | Time: " << end - start << " sec\n";
    total_sum = 0;
    start = omp_get_wtime();
    #pragma omp parallel for
    for (int i = 0; i < N; i++) {
        #pragma omp critical
        {
            total_sum += arr[i];
        }
    }

    end = omp_get_wtime();
    cout << "Critical Sum: " << total_sum
         << " | Time: " << end - start << " sec\n";
    delete[] arr;
    return 0;
}
